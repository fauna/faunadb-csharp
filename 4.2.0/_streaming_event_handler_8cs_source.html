24 public StreamingEventHandler(Stream dataSource, Func<Task> checkConnection = null)

28 streamReader = new StreamReader(dataSource);
29 if (checkConnection != null)

33
34 checkConnectionTask = Task.Factory.StartNew(async () =>
35 {
36 while (true)
37 {
38 if (token.IsCancellationRequested)
39 {
40 return;
41 }
42

44 {

47 catch (Exception ex)

53 }

55 Task.Delay(TIME_OUT_IN_MILLIS, token).Wait();
