279 public static DateTime FromIsoTime(string dateString, string format)
280 {
281 var dateTruncated = TruncateLastTwoDigits(dateString);
282 return DateTime.ParseExact(dateTruncated, format, CultureInfo.InvariantCulture, DateTimeStyles.AssumeUniversal | DateTimeStyles.AdjustToUniversal);
283 }
284
290 private static string TruncateLastTwoDigits(string iso)
291 {
292 var index = iso.LastIndexOf(".", StringComparison.CurrentCulture);
293
294 if (index >= 0)
295 {
296 iso = iso.Substring(0, Math.Min(iso.Length, index + 8));
297
298 if (!iso.EndsWith("Z", StringComparison.CurrentCulture))
299 {
300 iso += "Z";
301 }
302 }
303 else
304 {
305 if (iso.EndsWith("Z", StringComparison.CurrentCulture))
306 {
307 iso = iso.Substring(0, iso.Length - 1);
308 }
309
310 iso += ".0000000Z";
311 }
312
313 return iso;
314 }
